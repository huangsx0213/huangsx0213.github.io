import json
import pandas as pd
from utils.common_utils import CommonUtils
from utils.config_loader import ConfigLoader
from utils.logger import Logger
from utils.request_sender import RequestSender
from utils.request_body_builder import RequestBodyBuilder
from utils.request_headers_builder import RequestHeadersBuilder
from utils.response_handler import ResponseHandler
from utils.saved_fields_handler import SavedFieldsHandler
import os

class APITestRunner:
    def __init__(self, config_path='config/config.yaml'):
        # Initialize Logger first to log any errors during initialization
        self.logger = Logger()

        # Load configuration
        self.config = self.load_config(config_path)
        self.active_environment = self.config['environments'][self.config['active_environment']]
        self.base_url = self.active_environment['base_url']

        # Load other configurations
        self.endpoints = self.load_config('config/endpoints.yaml')
        self.headers_config = self.load_config('config/headers.yaml')
        self.body_defaults = self.load_config('config/body_defaults.yaml')
        self.templates = ConfigLoader.load_templates('config/body_templates')
        self.template_dir = 'config/body_templates'
        self.default_headers = self.headers_config['default']

        # Initialize other components
        self.saved_fields_manager = SavedFieldsHandler()
        self.body_builder = RequestBodyBuilder(self.template_dir, self.body_defaults, self.templates)
        self.headers_builder = RequestHeadersBuilder(self.headers_config)
        self.response_handler = ResponseHandler(self.logger)

    def run_tests(self, test_cases_path='test_cases/test_cases.xlsx', test_id=None):
        # Load test cases from Excel file
        try:
            test_cases = pd.read_excel(test_cases_path)
            self.logger.log("INFO", f"Successfully loaded test cases from {test_cases_path}")
        except Exception as e:
            self.logger.log("ERROR", f"Failed to load test cases from {test_cases_path}: {str(e)}")
            return

        # Execute each test case
        for index, row in test_cases.iterrows():
            if test_id and row['ID'] != test_id:
                continue
            self.execute_test_case(row, test_cases_path, index)

    def execute_test_case(self, test_case, test_cases_path, row_idx):
        try:
            self.logger.log("INFO", f"Starting execution of test case {test_case['ID']}")

            # Load saved fields
            saved_fields = self.saved_fields_manager.load_saved_fields()
            self.logger.log("INFO", f"Successfully loaded saved fields for test case {test_case['ID']}")

            # Replace placeholders with saved fields
            self.replace_placeholders_with_saved_fields(test_case, saved_fields, ['Body Modifications', 'Expected Result'])
            self.logger.log("INFO", f"Successfully replaced placeholders for test case {test_case['ID']}")

            # Format endpoint
            endpoint = self.get_formatted_endpoint(test_case, saved_fields)
            self.logger.log("INFO", f"Successfully formatted endpoint for test case {test_case['ID']}")

            # Prepare headers
            headers = self.headers_builder.prepare_headers(test_case['Headers'], saved_fields)
            self.logger.log("INFO", f"Successfully prepared headers for test case {test_case['ID']}")

            # Prepare request body
            body, format_type = self.prepare_request_body(test_case)
            self.logger.log("INFO", f"Successfully prepared body for test case {test_case['ID']}")

            # Log request
            method = test_case['Method']
            self.logger.log_request(test_case['ID'], test_case['Endpoint'], method, endpoint, headers, body, format_type)
            self.logger.log("INFO", f"Successfully logged request for test case {test_case['ID']}")

            # Send request and log response
            response = RequestSender.send_request(self.base_url, method, endpoint, headers, body, format_type)
            self.logger.log("INFO", f"Successfully sent request for test case {test_case['ID']}")
            self.logger.log_response(test_case['ID'], test_case['Endpoint'], response, format_type)
            self.logger.log("INFO", f"Successfully logged response for test case {test_case['ID']}")

            # Handle response
            self.response_handler.handle_response(response, test_case, test_cases_path, row_idx)
            self.logger.log("INFO", f"Successfully handled response for test case {test_case['ID']}")

            # Update saved fields
            self.saved_fields_manager.update_saved_fields(response, test_case)
            self.logger.log("INFO", f"Successfully updated saved fields for test case {test_case['ID']}")

            self.logger.log("INFO", f"Finished execution of test case {test_case['ID']}")
            self.logger.log("INFO", f"======================================================================\n")
        except Exception as e:
            self.logger.log("ERROR", f"Failed to execute test case {test_case['ID']}: {str(e)}")

    def get_formatted_endpoint(self, test_case, saved_fields):
        try:
            return self.endpoints[test_case['Endpoint']].format(**json.loads(test_case['Body Modifications']))
        except KeyError as e:
            self.logger.log("ERROR", f"Endpoint not found: {str(e)}")
            raise
        except json.JSONDecodeError as e:
            self.logger.log("ERROR", f"Failed to decode JSON for endpoint formatting: {str(e)}")
            raise

    def prepare_request_body(self, test_case):
        try:
            template_name = test_case['Template Name']
            template_defaults = test_case['Template Defaults']
            body_modifications = json.loads(test_case['Body Modifications'])
            default_values = self.body_defaults.get(template_defaults, {})
            format_type = CommonUtils.get_file_format(self.templates[template_name])
            body = self.body_builder.prepare_body(template_name, default_values, body_modifications, format_type)
            return body, format_type
        except KeyError as e:
            self.logger.log("ERROR", f"Template not found: {str(e)}")
            raise
        except json.JSONDecodeError as e:
            self.logger.log("ERROR", f"Failed to decode JSON for body preparation: {str(e)}")
            raise

    def replace_placeholders_with_saved_fields(self, test_case, saved_fields, columns):
        for key, value in saved_fields.items():
            for column in columns:
                if column in test_case and pd.notna(test_case[column]):
                    lines = test_case[column].splitlines()
                    replaced_lines = [line.replace(f"${{{key}}}", str(value)) for line in lines]
                    test_case[column] = "\n".join(replaced_lines)

    def load_config(self, file_path):
        try:
            return ConfigLoader.load_yaml(file_path)
        except Exception as e:
            self.logger.log("ERROR", f"Failed to load config from {file_path}: {str(e)}")
            raise

import os


class CommonUtils:
    @staticmethod
    def get_file_format(file_path):
        _, file_extension = os.path.splitext(file_path)
        return file_extension.lower().strip('.')

    @staticmethod
    def format_xml(xml_string):
        from lxml import etree
        parser = etree.XMLParser(remove_blank_text=True)
        xml_element = etree.fromstring(xml_string, parser)
        formatted_xml = etree.tostring(xml_element, pretty_print=False, encoding='unicode')
        return formatted_xml.strip()

import yaml
import json
import os


class ConfigLoader:

    @staticmethod
    def load_json(file_path):
        with open(file_path, 'r') as file:
            config = json.load(file)
        return config

    @staticmethod
    def load_yaml(file_path):
        with open(file_path, 'r') as file:
            config = yaml.safe_load(file)
        return config

    @staticmethod
    def load_templates(template_dir):
        templates = {}
        for filename in os.listdir(template_dir):
            if filename.endswith('.json') or filename.endswith('.xml'):
                template_name = os.path.splitext(filename)[0]
                templates[template_name] = os.path.join(template_dir, filename)
        return templates

import json
import os
import time
from datetime import datetime
from utils.common_utils import CommonUtils


class Logger:
    def __init__(self, log_dir='log'):
        self.log_dir = log_dir
        self.log_filename = self.generate_log_filename()
        self.log_filepath = os.path.join(self.log_dir, self.log_filename)
        if not os.path.exists(self.log_dir):
            os.makedirs(self.log_dir)

    def generate_log_filename(self):
        timestamp = int(time.time())
        date_str = datetime.fromtimestamp(timestamp).strftime('%Y-%m-%d_%H-%M-%S')
        return f"log_{date_str}.log"

    def log_request(self, test_id, endpoint_name, method, endpoint, headers, body, format_type):
        timestamp = int(time.time())
        date_str = datetime.fromtimestamp(timestamp).strftime('%Y-%m-%d_%H-%M-%S')
        if format_type == 'json':
            formatted_body = body
        elif format_type == 'xml':
            formatted_body = CommonUtils.format_xml(body)
        log_content1 = {
            "method": method,
            "endpoint": endpoint,
            "headers": headers,
            "body": formatted_body,
            "format_type": format_type
        }
        log_content2 = f"{date_str} [INFO] [Test ID: {test_id}] [Endpoint: {endpoint_name}]: Request:"
        with open(self.log_filepath, 'a') as log_file:
            log_file.write(log_content2 + "\n")
            json.dump(log_content1, log_file, indent=4)
            log_file.write("\n")

    def log_response(self, test_id, endpoint_name, response, format_type):
        timestamp = int(time.time())
        date_str = datetime.fromtimestamp(timestamp).strftime('%Y-%m-%d_%H-%M-%S')
        log_content1 = {
            "response_status_code": response.status_code,
            "response_headers": dict(response.headers),
            "response_body": json.loads(response.text) if format_type == 'json' else response.text
        }
        log_content2 = f"{date_str} [INFO] [Test ID: {test_id}] [Endpoint: {endpoint_name}]: Response:"
        with open(self.log_filepath, 'a') as log_file:
            log_file.write(log_content2 + "\n")
            json.dump(log_content1, log_file, indent=4)
            log_file.write("\n")

    def log(self, level, message):
        timestamp = int(time.time())
        date_str = datetime.fromtimestamp(timestamp).strftime('%Y-%m-%d_%H-%M-%S')
        log_content = f"{date_str} [{level.upper()}]: {message}"
        with open(self.log_filepath, 'a') as log_file:
            log_file.write(log_content + "\n")

import re
from .variable_generator import VariableGenerator
from .template_renderer import TemplateRenderer


class RequestBodyBuilder:
    def __init__(self, template_dir, body_defaults, templates):
        self.template_dir = template_dir
        self.body_defaults = body_defaults
        self.templates = templates

    def prepare_body(self, template_name, default_values, modifications, format_type):
        try:
            template_path = self.templates[template_name]
            # Merge default values and modifications
            body_data = self._merge_data(self._merge_dynamic_defaults(default_values), modifications)
            # Generate dynamic values
            dynamic_values = self._generate_dynamic_values(body_data)
            # Merge dynamic values
            body_data = self._merge_data(body_data, dynamic_values)
            return TemplateRenderer.render_template(self.template_dir, template_path, body_data, format_type)
        except KeyError as e:
            raise ValueError(f"Template '{template_name}' not found: {str(e)}")
        except Exception as e:
            raise ValueError(f"Error preparing request body: {str(e)}")

    def _merge_dynamic_defaults(self, default_values):
        merged_values = {}
        for key, value in default_values.items():
            if isinstance(value, dict):
                merged_values[key] = self._merge_dynamic_defaults(value)
            elif isinstance(value, str):
                merged_values[key] = self._process_string(value)
            else:
                merged_values[key] = value
        return merged_values

    def _generate_dynamic_values(self, nested_data):
        if isinstance(nested_data, dict):
            return {key: self._generate_dynamic_values(value) if isinstance(value, (dict, list))
                    else self._process_string(value) for key, value in nested_data.items()}
        elif isinstance(nested_data, list):
            return [self._generate_dynamic_values(item) if isinstance(item, (dict, list))
                    else self._process_string(item) for item in nested_data]
        return nested_data

    def _process_string(self, value):
        if isinstance(value, str) and re.match(r'\{\{\s*[^}]+?\s*\}\}', value):
            dynamic_key = re.findall(r'\{\{\s*([^}]+?)\s*\}\}', value)[0]
            return VariableGenerator.generate_dynamic_value(dynamic_key)
        return value

    def _merge_data(self, base_data, custom_data):
        for key, value in custom_data.items():
            if key in base_data and isinstance(value, dict) and isinstance(base_data[key], dict):
                base_data[key] = self._merge_data(base_data.get(key, {}), value)
            else:
                base_data[key] = value
        return base_data

import re
from .variable_generator import VariableGenerator

class RequestHeadersBuilder:
    def __init__(self, headers_config):
        self.headers_config = headers_config

    def prepare_headers(self, headers_name, saved_fields):
        try:
            headers = self.headers_config[headers_name]
            return {k: self._process_header_value(v, saved_fields) for k, v in headers.items()}
        except KeyError as e:
            raise ValueError(f"Headers configuration '{headers_name}' not found: {str(e)}")

    def _process_header_value(self, value, saved_fields):
        if isinstance(value, str) and re.match(r'\{\{\s*[^}]+?\s*\}\}', value):
            return self._generate_dynamic_value(value, saved_fields)
        return value

    def _generate_dynamic_value(self, template_string, saved_fields):
        dynamic_key = re.findall(r'\{\{\s*([^}]+?)\s*\}\}', template_string)[0]
        if dynamic_key in saved_fields:
            return saved_fields[dynamic_key]
        return VariableGenerator.generate_dynamic_value(dynamic_key)

import requests


class RequestSender:
    @staticmethod
    def send_request(base_url, method, endpoint, headers=None, body=None, format_type='json'):
        url = f"{base_url}{endpoint}"
        requests_method = RequestSender._get_request_method(method)

        if format_type == 'json':
            response = requests_method(url, headers=headers, json=body)
        elif format_type == 'xml':
            headers['Content-Type'] = 'application/xml'
            response = requests_method(url, headers=headers, data=body)
        else:
            raise ValueError(f"Unsupported format type: {format_type}")

        return response

    @staticmethod
    def _get_request_method(method):
        methods = {
            'GET': requests.get,
            'POST': requests.post,
            'PUT': requests.put,
            'DELETE': requests.delete,
            'PATCH': requests.patch
        }

        if method in methods:
            return methods[method]
        else:
            raise ValueError(f"Unsupported HTTP method: {method}")

import json
import re
import pandas as pd
from openpyxl.styles import Font
from openpyxl import load_workbook


class ResponseHandler:
    def __init__(self, logger):
        self.logger = logger

    def handle_response(self, response, test_case, test_cases_path, row_idx):
        actual_status = response.status_code
        actual_response = self._get_response_content(response)

        expected_lines = self._parse_lines(test_case['Expected Result'])
        fields_to_save_lines = self._parse_lines(test_case.get('Fields To Save', ''))

        results = [self._compare_field(actual_response, expectation) for expectation in expected_lines]
        fields_saved_results = [self._extract_field(actual_response, field) for field in fields_to_save_lines]

        overall_result = "fail" if any(res['result'] == "fail" for res in results) else "pass"
        formatted_results = self._format_results(results)
        formatted_fields_saved = self._format_saved_fields(fields_saved_results)

        self._write_results_to_excel(
            test_cases_path, row_idx + 2, actual_status, formatted_results, overall_result, formatted_fields_saved
        )

        self.logger.log("INFO", json.dumps(results, indent=4))
        self.logger.log("INFO", f"The overall result is: {overall_result}.")

    def _get_response_content(self, response):
        try:
            return response.json()
        except json.JSONDecodeError:
            return response.text

    def _parse_lines(self, lines):
        return lines.strip().split('\n') if pd.notna(lines) else []

    def _compare_field(self, actual_response, expectation):
        try:
            field_path, expected_value = expectation.split('=')
        except ValueError:
            return self._create_error_result(expectation, "Invalid [Expected Result] format")

        expected_value = expected_value.strip().strip('""')
        actual_value = self._get_actual_value(actual_response, field_path)

        if actual_value is None:
            return self._create_fail_result(field_path, "Field not found")
        return self._create_pass_fail_result(field_path, actual_value, expected_value)

    def _get_actual_value(self, actual_response, field_path):
        parts = field_path.split('.')
        try:
            if 'response[' in parts[0]:
                match = re.match(r'response\[(\d+)\]', parts[0])
                array_index = int(match.group(1))
                value = actual_response[array_index]
            else:
                value = actual_response

            for part in parts[1:]:
                if '[' in part and ']' in part:
                    array_part, idx = re.match(r'(.*)\[(\d+)\]', part).groups()
                    value = value[array_part][int(idx)]
                else:
                    value = value[part]
            return value
        except (KeyError, IndexError, TypeError):
            return None

    def _create_error_result(self, field_path, error_message):
        return {
            "field": field_path,
            "result": "error",
            "error": error_message
        }

    def _create_fail_result(self, field_path, error_message):
        return {
            "field": field_path,
            "result": "fail",
            "error": error_message
        }

    def _create_pass_fail_result(self, field_path, actual_value, expected_value):
        return {
            "field": field_path,
            "result": "pass" if str(actual_value) == expected_value else "fail"
        }

    def _extract_field(self, actual_response, field_path):
        parts = field_path.split('.')
        try:
            if 'response[' in parts[0]:
                match = re.match(r'response\[(\d+)\]', parts[0])
                array_index = int(match.group(1))
                value = actual_response[array_index]
            else:
                value = actual_response

            for part in parts[1:]:
                if '[' in part and ']' in part:
                    array_part, idx = re.match(r'(.*)\[(\d+)\]', part).groups()
                    value = value[array_part][int(idx)]
                else:
                    value = value[part]

            return f"{field_path}=\"{value}\"" if isinstance(value, str) else f"{field_path}={value}"
        except (KeyError, IndexError, TypeError):
            return f"{field_path}=null"

    def _format_results(self, results):
        return '\n'.join(f"{res['field']}:{res['result']}" for res in results)

    def _format_saved_fields(self, fields_saved_results):
        return '\n'.join(fields_saved_results)

    def _write_results_to_excel(self, file_path, row, actual_status, formatted_results, overall_result,
                                formatted_fields_saved):
        workbook = load_workbook(file_path)
        sheet = workbook.active
        actual_status_col_idx = self._get_column_index_by_name(sheet, "Actual Status")
        actual_result_col_idx = self._get_column_index_by_name(sheet, "Actual Result")
        overall_result_col_idx = self._get_column_index_by_name(sheet, "Overall Result")
        fields_saved_col_idx = self._get_column_index_by_name(sheet, "Fields Saved")

        sheet.cell(row=row, column=actual_status_col_idx, value=actual_status)
        sheet.cell(row=row, column=actual_result_col_idx, value=formatted_results)
        overall_cell = sheet.cell(row=row, column=overall_result_col_idx, value=overall_result)

        overall_cell.font = Font(color="00FF00" if overall_result == "pass" else "FF0000")
        sheet.cell(row=row, column=fields_saved_col_idx, value=formatted_fields_saved)

        workbook.save(file_path)

    def _get_column_index_by_name(self, sheet, column_name):
        for col in sheet.iter_cols(1, sheet.max_column):
            if col[0].value == column_name:
                return col[0].column
        return None

import re
import yaml
import os
import pandas as pd


class SavedFieldsHandler:
    def __init__(self, file_path='config/saved_fields.yaml'):
        self.file_path = file_path

    def load_saved_fields(self):
        if not os.path.exists(self.file_path):
            return {}
        try:
            with open(self.file_path, 'r') as f:
                saved_fields = yaml.safe_load(f) or {}
            return saved_fields
        except Exception as e:
            raise ValueError(f"Failed to load saved fields: {str(e)}")

    def save_fields(self, field_data):
        saved_fields = self.load_saved_fields()
        saved_fields.update(field_data)
        try:
            with open(self.file_path, 'w') as f:
                yaml.safe_dump(saved_fields, f, default_flow_style=False)
        except Exception as e:
            raise ValueError(f"Failed to save fields: {str(e)}")

    def update_saved_fields(self, response, test_case):
        fields_to_save_lines = self._parse_fields_to_save(test_case.get('Fields To Save', ''))
        actual_response = response.json()
        field_data = {
            f"{test_case['ID']}.{field}": self._extract_value(actual_response, field)
            for field in fields_to_save_lines
        }
        self.save_fields(field_data)

    def _parse_fields_to_save(self, fields_to_save):
        return fields_to_save.strip().split('\n') if pd.notna(fields_to_save) else []

    def _extract_value(self, actual_response, field_path):
        parts = field_path.split('.')
        try:
            if 'response[' in parts[0]:
                match = re.match(r'response\[(\d+)\]', parts[0])
                array_index = int(match.group(1))
                value = actual_response[array_index]
            else:
                value = actual_response

            for part in parts[1:]:
                if '[' in part and ']' in part:
                    array_part, idx = re.match(r'(.*)\[(\d+)\]', part).groups()
                    value = value[array_part][int(idx)]
                else:
                    value = value[part]
            return value
        except (KeyError, IndexError, TypeError):
            return None

import os
import json
from jinja2 import Environment, FileSystemLoader, TemplateNotFound


class TemplateRenderer:
    @staticmethod
    def render_template(template_dir, template_path, modifications, format_type):
        try:
            env = TemplateRenderer._create_environment(template_dir)
            template = TemplateRenderer._load_template(env, template_path)
            rendered_body = template.render(modifications)
            return TemplateRenderer._format_rendered_body(rendered_body, format_type)
        except (TemplateNotFound, json.JSONDecodeError) as e:
            raise ValueError(f"Error rendering template: {str(e)}")
        except Exception as e:
            raise ValueError(f"An unexpected error occurred while rendering template: {str(e)}")

    @staticmethod
    def _create_environment(template_dir):
        env = Environment(loader=FileSystemLoader(template_dir))
        env.filters['json_bool'] = lambda value: str(value).lower()
        return env

    @staticmethod
    def _load_template(env, template_path):
        template_name = os.path.basename(template_path)
        return env.get_template(template_name)

    @staticmethod
    def _format_rendered_body(rendered_body, format_type):
        if format_type == 'json':
            return json.loads(rendered_body)
        elif format_type == 'xml':
            return rendered_body
        else:
            raise ValueError(f"Unsupported format type: {format_type}")
